Uma função que recebe quantos parametros forem necessários e calcule a média 

       
    function media(xxx){
        var contagem = 0;
        var soma = 0;
        while (arguments[contagem]){
            soma += arguments[contagem];
            contagem++;
        }
        var results = soma / arguments.length;
        console.log(results);
    }
